【Tomcat構築手順書】
１）tar.gzファイルを/usr/local配下に配置
su - root
kenn0517
cd /usr/local/
cp -p /Users/user/Desktop/Develop/01_Construction/03_Middleware/apache-tomcat-8.0.36.tar.gz

２）tar.gzファイル解凍
tar zxvf apache-tomcat-8.0.36.tar.gz

３）シンボリックリンク作成
cd /Library
ln -s /usr/local/apache-tomcat-8.0.36 Tomcat

４）起動シェル権限変更
chmod +x /Library/Tomcat/bin/*.sh
chmod 644 /usr/local/apache-tomcat-8.0.36/conf/*

５）アカウント・パスワード設定
アカウント：admin
パスワード：admin
/usr/local/apache-tomcat-8.0.36/conf
cp -p tomcat-users.xml tomcat-users.xml_bk
vi tomcat-users.xml
↓下記追加
  <role rolename="manager-gui"/>
  <user username="admin" password="admin" roles="manager-gui"/>

６）起動
cd /Library/Tomcat/bin/
./startup.sh
